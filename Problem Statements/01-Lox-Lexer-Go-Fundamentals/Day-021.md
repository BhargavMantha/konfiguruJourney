**2. File Scanning Testing:**
```bash
# Test all examples
for file in examples/*.lox; do
    echo "Testing $file"
    ./lox "$file"
done

# Test with verbose mode
./lox -v examples/fibonacci.lox

# Test error handling
echo 'var x = "unterminated' | ./lox
```

**3. Edge Cases:**
```bash
# Empty file
touch /tmp/empty.lox
./lox /tmp/empty.lox

# Very long line
echo "var x = $(printf 'a%.0s' {1..1000});" > /tmp/long.lox
./lox /tmp/long.lox

# File doesn't exist
./lox nonexistent.lox
```
